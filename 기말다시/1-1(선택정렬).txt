#include <stdio.h>
#include <stdlib.h>
//selection Sort
void printArray(int a[], int size) // 필요하면 쓴다
{
	int i;
	for (i = 0; i < size; i++)
		printf("%d ", a[i]);
	printf("\n");
}
void rangeArray(int a[], int n)
{
   int i, j, min, minIdx, temp;

	for (i = 0; i < n - 1; i++) {
		//max찾기
		min = a[i];
		minIdx = i;
		for (j = i; j <= n - 1; j++) {
			if (min <= a[j]) {
				min = a[j];
				minIdx = j;
			}
		}
		temp = a[i];
		a[i] = min;
		a[minIdx] = temp;
	}
}
int main(void)
{
	int student, test;
	int i, j, input;
	int *stSub;
    
	scanf("%d", &test);
	scanf("%d", &student);

	stSub = (int *)malloc(sizeof(int) * student);

	for (i = 0; i < student; i++) {
		stSub[i] = 0;
	}

	for (i = 0; i < test; i++)
		for (j = 0; j < student; j++) {
			scanf("%d", &input);
            stSub[j] += input;
		}
        
    //printArray(stSub, student);
    rangeArray(stSub, student);
    printArray(stSub, student);
}
